\begin{algorithm}[H]
    \renewcommand{\thealgorithm}{}
    \caption{``签到”加工小说明} 
    \label{alg3} 
    \begin{algorithmic}[1]
        \IF{Pass Check 参与人员名单 then}
        \IF{Pass Check 签到信息 then}
        \STATE Generate 签到时间+签到地点 Based on 签到信息
        \STATE Write 志愿者ID+志愿项目ID+签到时间+签到地点 to 签到人员名单
        
        \ELSE return 签到失败信息
        \ENDIF
        \ELSE return 签到失败信息
        \ENDIF
    \end{algorithmic} 
\end{algorithm}
