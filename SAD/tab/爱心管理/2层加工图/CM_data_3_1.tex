    \renewcommand{\thealgorithm}{}
    \caption{“爱心捐款”加工小说明} 
    \label{alg3} 
    \begin{algorithmic}[1] 
        \STATE Select Sum（资助捐款）as 项目资助 Group by 志愿项目
